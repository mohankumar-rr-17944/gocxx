See It In Action

Channels & Concurrency

channels.cpp
#include <gocxx/gocxx.h>
using namespace gocxx::base;

int main() {
    // Create buffered channel
    Chan<int> ch(5);
    
    // Producer thread
    std::thread producer([&ch]() {
        for (int i = 0; i < 10; ++i) {
            ch << i;  // Send to channel
        }
        ch.close();
    });
    
    // Consumer
    int value;
    while (ch >> value) {
        std::cout << "Received: " << value;
    }
    
    producer.join();
}

Select & Defer

select.cpp
// Select from multiple channels
select(
    recv_case(ch1, [](auto val) {
        std::cout << "Ch1: " << val;
    }),
    recv_case(ch2, [](auto val) {
        std::cout << "Ch2: " << val;
    }),
    default_case([]() {
        std::cout << "No channel ready";
    })
);

// Automatic cleanup with defer
{
    auto* resource = new int(42);
    defer([resource]() {
        delete resource;
        std::cout << "Cleaned up!";
    });
    
    // Use resource...
    // Automatic cleanup on scope exit
}

Get Started in Minutes

1

Clone Repository

Get the latest version from GitHub

git clone https://github.com/gocxx/gocxx.git
2

Build with CMake

Standard CMake build process

mkdir build && cd build cmake .. -DGOCXX_ENABLE_TESTS=ON cmake --build .
3

Include in Project

Add to your CMakeLists.txt

add_subdirectory(gocxx) target_link_libraries(myapp PRIVATE gocxx)
